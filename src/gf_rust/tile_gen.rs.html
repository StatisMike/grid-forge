25/// Class handling the generation of the tilemap.
26///
27/// The class is responsible for the generation of the [`GridMap2D`], and trasferring its results to the Godot's [`TileMap`]
28/// afterwards.
29///
30/// The generation itself is done in a separate thread, so the main thread will not be blocked.
31#[derive(GodotClass)]
32#[class(base=Node, init)]
33pub struct TileGenerator {
34    /// The collection of tiles to be used for the generation.
35    #[export]
36    collection: Option<Gd<TileCollections>>,
37    /// Modal dialog to be shown when the generation encounters an error.
38    #[export]
39    modal: Option<Gd<AcceptDialog>>,
40    /// Dictionary of pregenerated tiles, with their [`Vector2i`] position as a key and [`SingleTile`] as a value.
41    #[var]
42    pregenerated: Dictionary,
43
44    /// Produced generation history.
45    generation_history: Vec<singular::CollapseHistoryItem>,
46    /// Produced GridMap2D.
47    generated: Option<GridMap2D<BasicIdentTileData>>,
48
49    // ---------------- Generation stats -------------- //
50    /// Total time spent on the generation, in microseconds.
51    #[var]
52    generation_time_us_total: u32,
53    /// Time spent on the generation for successful run, in microseconds.
54    #[var]
55    generation_time_us_success: u32,
56
57    // ---------------- Generation state -------------- //
58    running: bool,
59
60    // ---------------- Second thread utils -------------- //
61    handle: Option<JoinHandle<()>>,
62    channel: Option<Receiver<GenerationResult>>,
63
64    // ---------- Collapsible generation rules ----------- //
65    border_rules: singular::AdjacencyRules<BasicIdentTileData>,
66    identity_rules: singular::AdjacencyRules<BasicIdentTileData>,
67    frequency_hints: singular::FrequencyHints<BasicIdentTileData>,
68
69    base: Base<Node>,
70}

72#[godot_api]
73impl INode for TileGenerator {
74    /// Process retrieves the signals from spawned thread when the generation is running, emitting them back to the
75    /// Godot's *MainNode* on the main thread.
76    fn process(&mut self, _delta: f64) {
77        if !self.running {
78            return;
79        }
80
81        if let Some(receiver) = &self.channel {
82            if let Ok(result) = receiver.try_recv() {
83                match result {
84                    // Runtime error occured - only passing the error message to Godot, generator will retry.
85                    GenerationResult::RuntimeErr(mssg) => {
86                        self.base_mut().emit_signal(
87                            "generation_runtime_error",
88                            &[GString::from(mssg).to_variant()],
89                        );
90                    }
91
92                    // Fatal error occured, the generation will be stopped.
93                    GenerationResult::Error(mssg) => {
94                        self.base_mut()
95                            .emit_signal("generation_error", &[GString::from(mssg).to_variant()]);
96                        self.base_mut()
97                            .emit_signal("generation_finished", &[false.to_variant()]);
98                        if self.handle.is_some() {
99                            let thread = self.handle.take();
100                            thread.unwrap().join().unwrap();
101                        }
102                        self.running = false;
103                    }
104
105                    // Successful generation - the information about the success will be sent to the Godot's *MainNode*,
106                    // while all of the generated data will be accessible using other Godot-exposed methods.
107                    GenerationResult::Success((
108                        map,
109                        history,
110                        (duration_total, duration_success),
111                    )) => {
112                        (self.generated, self.generation_history) = (Some(map), history);
113                        self.generation_time_us_total = duration_total.as_micros() as u32;
114                        self.generation_time_us_success = duration_success.as_micros() as u32;
115                        self.base_mut()
116                            .emit_signal("generation_finished", &[true.to_variant()]);
117                        if self.handle.is_some() {
118                            let thread = self.handle.take();
119                            thread.unwrap().join().unwrap();
120                        }
121                        self.running = false;
122                    }
123                }
124            }
125        }
126    }
127}

129#[godot_api]
130impl TileGenerator {
131    /// Emitted when the generation is finished.
132    #[signal]
133    fn generation_finished(success: bool);
134
135    /// Emitted when the generation encounters an error and will be stopped.
136    #[signal]
137    fn generation_error(message: GString);
138
139    /// Emitted if the generation encounters an error, but will retry.
140    #[signal]
141    fn generation_runtime_error(message: GString);
142
143    #[constant]
144    const IDENTITY_RULES: i32 = 0;
145    #[constant]
146    const ADJACENCY_RULES: i32 = 1;
147
148    #[constant]
149    const POSITION_QUEUE: i32 = 0;
150    #[constant]
151    const ENTROPHY_QUEUE: i32 = 1;
152
153    /// Initializes the single-tiled rulesets for the generation.
154    #[func]
155    fn initialize_rulesets(&mut self, maps: Array<GString>) {
156        let mut grid_maps = Vec::new();
157        for map_path in maps.iter_shared() {
158            match self
159                .collection
160                .as_ref()
161                .unwrap()
162                .bind()
163                .load_vis_map_from_path(&map_path.to_string())
164            {
165                Ok(map) => {
166                    grid_maps.push(map);
167                }
168                Err(err) => {
169                    self.show_modal(&format!("Error loading map for TileGenerator: {err}"));
170                    return;
171                }
172            }
173        }
174
175        let mut analyzer = singular::BorderAnalyzer::default();
176        for map in grid_maps.iter() {
177            analyzer.analyze(map);
178        }
179        self.border_rules = analyzer.adjacency().clone();
180
181        let mut analyzer = singular::IdentityAnalyzer::default();
182        for map in grid_maps.iter() {
183            analyzer.analyze(map)
184        }
185        self.identity_rules = analyzer.adjacency().clone();
186
187        let mut frequency_hints = singular::FrequencyHints::default();
188
189        for map in grid_maps.iter() {
190            frequency_hints.analyze(map);
191        }
192
193        self.frequency_hints = frequency_hints;
194    }
195
196    /// Starts the generation of the tilemap. Whole generation will be done in a separate thread, so the main thread will not be blocked.
197    #[func]
198    fn begin_generation(
199        &mut self,
200        width: i32,
201        height: i32,
202        rule: i32,
203        queue: i32,
204        pregenerated: Dictionary,
205    ) {
206        let (sender, receiver) = mpsc::channel();
207        let subscriber = singular::CollapseHistorySubscriber::default();
208        let mut collapsed_grid = None;
209        self.channel = Some(receiver);
210        self.running = true;
211        let size = GridSize::new_xy(width as u32, height as u32);
212
213        let collapsed = self.unpack_pregenerated(&pregenerated);
214
215        if !collapsed.is_empty() {
216            let mut grid = CollapsedGrid::new(size);
217            for (pos, single_tile) in collapsed.iter() {
218                grid.insert_tile(GridTile::new(
219                    *pos,
220                    CollapsedTileData::new(single_tile.bind().get_tile_type_id() as u64),
221                ));
222            }
223            collapsed_grid = Some(grid);
224        }
225        self.pregenerated = pregenerated;
226
227        let adjacency_rules = if rule == Self::ADJACENCY_RULES {
228            self.border_rules.clone()
229        } else {
230            self.identity_rules.clone()
231        };
232
233        let frequency_hints = self.frequency_hints.clone();
234
235        let builder = IdentTileTraitBuilder::<BasicIdentTileData>::default();
236
237        self.handle = Some(thread::spawn(move || {
238            const RETRY_COUNT: usize = 10;
239
240            let mut iter = 0;
241            let mut rng = thread_rng();
242            let mut resolver = singular::Resolver::default().with_subscriber(Box::new(subscriber));
243
244            let mut grid =
245                singular::CollapsibleTileGrid::new_empty(size, &frequency_hints, &adjacency_rules);
246
247            if let Some(collapsed) = collapsed_grid {
248                if let Err(err) = grid.populate_from_collapsed(&collapsed) {
249                    sender
250                        .send(GenerationResult::Error(format!(
251                            "Cannot populate grid from collapsed grid: {err}"
252                        )))
253                        .unwrap();
254                    return;
255                }
256            }
257
258            let empty_positions = grid.empty_positions();
259
260            let mut duration_total: Duration;
261            let mut duration_success: Duration;
262
263            let start_total = Instant::now();
264
265            loop {
266                let start_success = Instant::now();
267                let result = if queue == Self::POSITION_QUEUE {
268                    resolver.generate_position(
269                        &mut grid,
270                        &mut rng,
271                        &empty_positions,
272                        PositionQueue::default(),
273                    )
274                } else {
275                    resolver.generate_entrophy(&mut grid, &mut rng, &empty_positions)
276                };
277                duration_total = start_total.elapsed();
278                duration_success = start_success.elapsed();
279                match result {
280                    Ok(_) => break,
281                    Err(err) => {
282                        if iter >= RETRY_COUNT {
283                            sender
284                                .send(GenerationResult::Error(format!(
285                                    "Cannot generate after {RETRY_COUNT} tries: {err}"
286                                )))
287                                .unwrap();
288                            return;
289                        }
290                        sender
291                            .send(GenerationResult::RuntimeErr(format!(
292                                "Error in iter: {iter}: {err}"
293                            )))
294                            .unwrap();
295                        iter += 1;
296                    }
297                }
298            }
299
300            let map = grid.retrieve_ident(&builder).unwrap();
301            let subscriber = resolver.retrieve_subscriber().unwrap();
302            let history = subscriber
303                .as_any()
304                .downcast_ref::<singular::CollapseHistorySubscriber>()
305                .unwrap()
306                .history()
307                .to_vec();
308            sender
309                .send(GenerationResult::Success((
310                    map,
311                    history,
312                    (duration_total, duration_success),
313                )))
314                .unwrap();
315        }));
316    }

318    /// Transfers the generated [`GridMap2D`] to the Godot's [`TileMap`].
319    #[func]
320    fn generated_to_tilemap(&self, tilemap: Gd<TileMap>) -> bool {
321        let mut tilemap = tilemap.clone();
322        if let Some(map) = &self.generated {
323            self.collection
324                .as_ref()
325                .unwrap()
326                .bind()
327                .grid_map_into_tilemap(map, &mut tilemap)
328                .unwrap();
329            return true;
330        }
331        false
332    }

334    fn show_modal(&self, message: &str) {
335        if let Some(modal) = &self.modal {
336            let mut pntr = modal.clone();
337            pntr.set_text(message);
338            pntr.set_visible(true);
339        } else {
340            godot_warn!("Cannot find modal for TileGenerator. Message to show: {message}");
341        }
342    }
343
344    fn get_generation_result(
345        &self,
346    ) -> Option<(
347        &GridMap2D<BasicIdentTileData>,
348        &Vec<singular::CollapseHistoryItem>,
349    )> {
350        if let Some(map) = &self.generated {
351            return Some((map, &self.generation_history));
352        }
353        None
354    }
355
356    fn unpack_pregenerated(
357        &self,
358        pregenerated: &Dictionary,
359    ) -> Vec<(GridPosition, Gd<SingleTile>)> {
360        let mut out = Vec::new();
361        for (key, value) in pregenerated.iter_shared() {
362            let pos = Vector2i::from_variant(&key);
363            let tile = Gd::<SingleTile>::from_variant(&value);
364            out.push((GridPosition::new_xy(pos.x as u32, pos.y as u32), tile));
365        }
366        out
367    }

370/// Result of the [`TileGenerator`] generation, received from the spawned thread, to be send into Godot's *MainNode*.
371enum GenerationResult {
372    /// Runtime error - only passing the error message to Godot, generator will retry.
373    RuntimeErr(String),
374    /// Fatal error, the generation will be stopped.
375    Error(String),
376    /// Successful generation - the generated map will be sent to Godot's *MainNode*.
377    Success(
378        (
379            GridMap2D<BasicIdentTileData>,
380            Vec<singular::CollapseHistoryItem>,
381            (Duration, Duration),
382        ),
383    ),
384}

386/// Class storing the history of the generation, providing the methods to view it in the Godot scene.
387#[derive(GodotClass)]
388#[class(base=Node, init)]
389pub struct GenerationHistoryState {
390    playing: bool,
391    timer: Option<Gd<Timer>>,
392    #[export]
393    tilemap: Option<Gd<TileMap>>,
394    #[export]
395    collection: Option<Gd<TileCollections>>,
396    history: Vec<singular::CollapseHistoryItem>,
397    #[var]
398    current: u32,
399    #[var]
400    total: u32,
401    base: Base<Node>,
402}
403
404impl GenerationHistoryState {
405    fn add_from_current(&mut self) -> bool {
406        if self.current >= self.total {
407            return false;
408        };
409        let (Some(map), Some(collection)) = (&mut self.tilemap, &self.collection) else {
410            godot_error!("Cannot draw frame, because either tilemap or collection is not set");
411            return false;
412        };
413        let Some(item) = self.history.get(self.current as usize - 1) else {
414            godot_error!("Cannot draw frame, because history is empty");
415            return false;
416        };
417        collection.bind().insert_tile(
418            map.clone(),
419            item.tile_type_id,
420            item.position.get_godot_coords(),
421        );
422        true
423    }
424
425    fn remove_from_current(&mut self) -> bool {
426        if self.current == 0 {
427            return false;
428        }
429        let Some(map) = &mut self.tilemap else {
430            godot_error!("Cannot draw frame, because either tilemap or collection is not set");
431            return false;
432        };
433        let Some(item) = self.history.get(self.current as usize - 1) else {
434            godot_error!("Cannot draw frame, because history is empty");
435            return false;
436        };
437        map.set_cell(0, item.position.get_godot_coords());
438        true
439    }
440}

442#[godot_api]
443impl GenerationHistoryState {
444    /// Sends the current step of the generation history to other Godot classes.
445    #[signal]
446    fn current_state(current: u32);
447
448    /// Retrieves the history of the generation from the [`TileGenerator`]
449    #[func]
450    pub fn set_history_from_generator(&mut self, generator: Gd<TileGenerator>) {
451        let binding = generator.bind();
452        let Some((gridmap, history)) = binding.get_generation_result() else {
453            godot_error!("Cannot get generation result from generator");
454            return;
455        };
456
457        if let Some(map) = &mut self.tilemap {
458            map.call(
459                "adjust_generation",
460                &[
461                    Vector2i::new(gridmap.size().x() as i32, gridmap.size().y() as i32)
462                        .to_variant(),
463                ],
464            );
465        }
466        self.history.clone_from(history);
467        self.current = 0;
468        self.total = self.history.len() as u32 + 1;
469        self.base_mut()
470            .emit_signal("current_state", &[0.to_variant()]);
471    }
472
473    /// Plays the generation history starting from the current step with the given speed.
474    #[func]
475    fn play(&mut self, count_per_sec: u32) {
476        if self.current >= self.total {
477            return;
478        }
479        self.playing = true;
480        let mut timer = Timer::new_alloc();
481        self.base_mut().add_child(&timer);
482        timer.set_wait_time(1. / count_per_sec as f64);
483        timer.connect("timeout", &self.base_mut().callable("forward"));
484        timer.start();
485        self.timer = Some(timer);
486    }
487
488    /// Stops the generation history playback.
489    #[func]
490    fn stop(&mut self) {
491        self.playing = false;
492        if self.timer.is_some() {
493            let mut timer = self.timer.take().unwrap();
494            timer.stop();
495            timer.queue_free();
496        }
497    }
498
499    /// Advances the generation history to the next step manually.
500    #[func]
501    fn forward(&mut self) {
502        if self.current >= self.total {
503            return;
504        }
505        self.current += 1;
506        if !self.add_from_current() && self.timer.is_some() {
507            let mut timer = self.timer.take().unwrap();
508            timer.stop();
509            timer.queue_free();
510            self.playing = false;
511        }
512        let current = self.current;
513        self.base_mut()
514            .emit_signal("current_state", &[current.to_variant()]);
515    }
516
517    /// Rewinds the generation history to the previous step.
518    #[func]
519    fn backward(&mut self) {
520        self.remove_from_current();
521        if self.current == 0 {
522            return;
523        }
524        self.current -= 1;
525        let current = self.current;
526        self.base_mut()
527            .emit_signal("current_state", &[current.to_variant()]);
528    }
529
530    /// Rewinds the generation history completely.
531    #[func]
532    fn rewind(&mut self) {
533        self.remove_from_current();
534        while self.current > 0 {
535            self.current -= 1;
536            self.remove_from_current();
537        }
538    }
