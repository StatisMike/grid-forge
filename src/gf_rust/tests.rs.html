17const TILESET_RESOURCE_PATH: &str = "res://tileset.tres";
18const SOURCE_ID: i32 = 0;
19const TILESET_IMG_PATH: &str = "res://tiles/all_tiles.png";

21const ROADS_MAP_PATH: &str = "res://tiles/roads.png";
22const SEAS_MAP_PATH: &str = "res://tiles/seas.png";

24#[gditest]
25fn test_load_into_gridmap() {
26    let collection = get_test_collection();
27    let cloned = collection.clone();
28    let binding = collection.bind();
29    let tileset = load::<TileSet>(TILESET_RESOURCE_PATH);
30
31    let roads_map = binding.load_vis_map_from_path(ROADS_MAP_PATH).unwrap();
32    let mut roads_tilemap = TileMap::new_alloc();
33    roads_tilemap.set_tileset(&tileset);
34    write_gridmap_to_tilemap(
35        &roads_map,
36        &mut roads_tilemap,
37        binding.godot_collection.as_ref().unwrap(),
38    )
39    .expect("cannot write roads to tilemap");
40
41    let seas_map = binding.load_vis_map_from_path(SEAS_MAP_PATH).unwrap();
42    let mut seas_tilemap = TileMap::new_alloc();
43    seas_tilemap.set_tileset(&tileset);
44    write_gridmap_to_tilemap(
45        &seas_map,
46        &mut seas_tilemap,
47        binding.godot_collection.as_ref().unwrap(),
48    )
49    .expect("cannot write seas to tilemap");
50
51    assert_eq!(
52        roads_map.size().x(),
53        size_from_rect(roads_tilemap.get_used_rect()).x()
54    );
55    assert_eq!(
56        roads_map.size().y(),
57        size_from_rect(roads_tilemap.get_used_rect()).y()
58    );
59    assert_eq!(
60        seas_map.size().x(),
61        size_from_rect(seas_tilemap.get_used_rect()).x()
62    );
63    assert_eq!(
64        seas_map.size().y(),
65        size_from_rect(seas_tilemap.get_used_rect()).y()
66    );
67
68    std::mem::drop(binding);
69
70    roads_tilemap.free();
71    seas_tilemap.free();
72    cloned.free()
73}

75#[gditest]
76fn test_from_grindmap_identical() {
77    let collection = get_test_collection();
78    let cloned = collection.clone();
79    let binding = collection.bind();
80    let builder = IdentTileTraitBuilder::<BasicIdentTileData>::default();
81    let tileset = load::<TileSet>(TILESET_RESOURCE_PATH);
82
83    let roads_map = binding.load_vis_map_from_path(ROADS_MAP_PATH).unwrap();
84    let mut roads_tilemap = TileMap::new_alloc();
85    roads_tilemap.set_tileset(&tileset);
86    write_gridmap_to_tilemap(
87        &roads_map,
88        &mut roads_tilemap,
89        binding.godot_collection.as_ref().unwrap(),
90    )
91    .expect("cannot write roads to tilemap");
92
93    let seas_map = binding.load_vis_map_from_path(SEAS_MAP_PATH).unwrap();
94    let mut seas_tilemap = TileMap::new_alloc();
95    seas_tilemap.set_tileset(&tileset);
96    write_gridmap_to_tilemap(
97        &seas_map,
98        &mut seas_tilemap,
99        binding.godot_collection.as_ref().unwrap(),
100    )
101    .expect("cannot write seas to tilemap");
102
103    let second_roads = load_gridmap_from_tilemap_manual(
104        &roads_tilemap,
105        binding.godot_collection.as_ref().unwrap(),
106        &builder,
107    )
108    .expect("cannot load roads to gridmap");
109    let second_seas = load_gridmap_from_tilemap_manual(
110        &seas_tilemap,
111        binding.godot_collection.as_ref().unwrap(),
112        &builder,
113    )
114    .expect("cannot load roads to gridmap");
115
116    for position in roads_map.get_all_positions() {
117        assert_eq!(
118            roads_map
119                .get_tile_at_position(&position)
120                .unwrap()
121                .as_ref()
122                .tile_type_id(),
123            second_roads
124                .get_tile_at_position(&position)
125                .unwrap()
126                .as_ref()
127                .tile_type_id()
128        );
129    }
130
131    for position in seas_map.get_all_positions() {
132        assert_eq!(
133            seas_map
134                .get_tile_at_position(&position)
135                .unwrap()
136                .as_ref()
137                .tile_type_id(),
138            second_seas
139                .get_tile_at_position(&position)
140                .unwrap()
141                .as_ref()
142                .tile_type_id()
143        );
144    }
145
146    std::mem::drop(binding);
147
148    roads_tilemap.free();
149    seas_tilemap.free();
150    cloned.free();
151}

153fn get_test_collection() -> Gd<TileCollections> {
154    let mut collection = TileCollections::new_alloc();
155    let out = collection.clone();
156
157    let tileset = load::<TileSet>(TILESET_RESOURCE_PATH);
158
159    let mut bind = collection.bind_mut();
160    bind.set_path_to_image(TILESET_IMG_PATH.into());
161    bind.set_source_id(SOURCE_ID);
162    bind.set_tileset(Some(tileset));
163    bind.generate_collections();
164
165    out
166}

168fn size_from_rect(rect: Rect2i) -> GridSize {
169    GridSize::new_xy(
170        (rect.size.x - rect.position.x) as u32,
171        (rect.size.y - rect.position.y) as u32,
172    )
