15/// Easily load [`GridMap2D`] of [`IdentifiableTileData`]-implementing TileData, automatically saving each tile into provided
16/// [`VisCollection`].
17///
18/// # Arguments
19/// - `image_buffer` - an [`ImageBuffer`] containing the source image data.
20/// - `collection` - a [`VisCollection`] containing the loaded tile pixels. New tile pixels will be automatically
21///   inserted, calculating the `tile_type_id` on basis of tile pixels.
22/// - `builder` - a struct which can be used to construct new tiles on basis of their `tile_id`. One of [`IdentTileBuilder`]
23///   implementing objects.
24///
25/// # Warning
26/// As the `tile_type_id` **is automatically calculated** with this function on basis of pixels, it won't work with specific,
27/// manually declared identifiers. In this case, you need to use [`load_gridmap_identifiable_manual`].
28pub fn load_gridmap_identifiable_auto<Data, P, B, const WIDTH: usize, const HEIGHT: usize>(
29    image_buffer: &ImageBuffer<P, Vec<P::Subpixel>>,
30    collection: &mut VisCollection<P, WIDTH, HEIGHT>,
31    builder: &B,
32) -> Result<GridMap2D<Data>, VisError<WIDTH, HEIGHT>>
33where
34    Data: IdentifiableTileData,
35    P: PixelWithDefault + 'static,
36    B: IdentTileBuilder<Data>,
37{
38    let size = check_grid_vis_size(image_buffer)?;
39    let mut grid = GridMap2D::<Data>::new(size);
40
41    for position in size.get_all_possible_positions() {
42        let pixels = VisCollection::<P, WIDTH, HEIGHT>::read_pixels_for_tile_at_pos(
43            image_buffer,
44            &position,
45        )?;
46        let tile = builder.build_tile_unchecked(position, create_tile_id_from_pixels(&pixels));
47        match collection.add_tile_pixels(&tile, image_buffer)? {
48            super::collection::VisCollectionOutcome::Empty => {
49                continue;
50            }
51            _ => grid.insert_tile(tile),
52        };
53    }
54    Ok(grid)
55}

57/// Load [`GridMap2D`] of [`IdentifiableTileData`]-implementing struct using registered pixels in [`VisCollection`].
58///
59/// If you don't need to declare the `tile_type_id` yourself, you can use [`load_gridmap_identifiable_auto`], which generates
60/// identifiers on basis of each pixel.
61///
62/// # Arguments
63/// - `image_buffer` - an [`ImageBuffer`] containing the source image data.
64/// - `collection` - a [`VisCollection`] containing the tile pixels. It needs to be populated before usage.
65/// - `builder` - a struct which can be used to construct new tiles on basis of their `tile_id`. One of [`IdentTileBuilder`]
66///   implementing objects.
67pub fn load_gridmap_identifiable_manual<Data, P, B, const WIDTH: usize, const HEIGHT: usize>(
68    image_buffer: &ImageBuffer<P, Vec<P::Subpixel>>,
69    collection: &VisCollection<P, WIDTH, HEIGHT>,
70    builder: &B,
71) -> Result<GridMap2D<Data>, VisError<WIDTH, HEIGHT>>
72where
73    Data: IdentifiableTileData,
74    P: PixelWithDefault + 'static,
75    B: IdentTileBuilder<Data>,
76{
77    let size = check_grid_vis_size(image_buffer)?;
78    let mut grid = GridMap2D::<Data>::new(size);
79
80    for position in size.get_all_possible_positions() {
81        let pixels = VisCollection::<P, WIDTH, HEIGHT>::read_pixels_for_tile_at_pos(
82            image_buffer,
83            &position,
84        )?;
85        if let Some(tile_id) = collection.get_tile_id_by_pixels(&pixels) {
86            grid.insert_tile(builder.build_tile_unchecked(position, *tile_id));
87        } else if !collection.is_empty(&pixels) {
88            return Err(VisError::new_nonexist(position));
89        }
90    }
91    Ok(grid)
92}

94/// Utility function to generate [`ImageBuffer`] of correct size for specific size of [`GridMap2D`] to write into
95/// with [`write_gridmap_identifiable`] and [`write_gridmap_vis`].
96pub fn init_map_image_buffer<P, const WIDTH: usize, const HEIGHT: usize>(
97    grid_size: &GridSize,
98) -> ImageBuffer<P, Vec<P::Subpixel>>
99where
100    P: PixelWithDefault,
101{
102    ImageBuffer::new(grid_size.x() * WIDTH as u32, grid_size.y() * HEIGHT as u32)
103}

105/// Write [`GridMap2D`] comprised of tiles containing [`IdentifiableTileData`] into provided [`ImageBuffer`], using
106/// pixel data gatheed in [`VisCollection`].
107///
108/// To make sure that the image buffer has exact correct size needed to write the `GridMap2D` representation into it,
109/// you can use [`check_grid_image_size`] beforehand.
110pub fn write_gridmap_identifiable<Data, P, const WIDTH: usize, const HEIGHT: usize>(
111    image_buffer: &mut ImageBuffer<P, Vec<P::Subpixel>>,
112    grid_map: &GridMap2D<Data>,
113    collection: &VisCollection<P, WIDTH, HEIGHT>,
114) -> Result<(), VisError<WIDTH, HEIGHT>>
115where
116    Data: IdentifiableTileData,
117    P: PixelWithDefault + 'static,
118{
119    collection.draw_map(grid_map, image_buffer)?;
120
121    Ok(())
122}

124/// Write [`GridMap2D`] comprised of tiles containing [`VisTileData`] into provided [`ImageBuffer`]. Pixel data retrieved
125/// via [`VisTileData::vis_pixels`] will be used.
126///
127/// To make sure that the image buffer has exact correct size needed to write the `GridMap2D` representation into it,
128/// you can use [`check_grid_image_size`] beforehand.
129pub fn write_gridmap_vis<Data, P, const WIDTH: usize, const HEIGHT: usize>(
130    image_buffer: &mut ImageBuffer<P, Vec<P::Subpixel>>,
131    grid_map: &GridMap2D<Data>,
132) -> Result<(), VisError<WIDTH, HEIGHT>>
133where
134    Data: VisTileData<P, WIDTH, HEIGHT>,
135    P: PixelWithDefault + 'static,
136{
137    for position in grid_map.get_all_positions() {
138        grid_map
139            .get_tile_at_position(&position)
140            .expect("cannot get tile")
141            .vis_to_buffer(image_buffer)?;
142    }
143
144    Ok(())
145}

147/// Checks the size of the [`ImageBuffer`] while loading [`GridMap2D`] from its visual representation, and produces
148/// the [`GridSize`] inferred from the image size. Results in [`VisError`] if the image size is not compatible
149/// with provided tile size in pixels.
150pub fn check_grid_vis_size<P: Pixel + 'static, const WIDTH: usize, const HEIGHT: usize>(
151    image: &ImageBuffer<P, Vec<P::Subpixel>>,
152) -> Result<GridSize, VisError<WIDTH, HEIGHT>> {
153    if image.height() as usize % HEIGHT != 0 || image.width() as usize % WIDTH != 0 {
154        Err(VisError::new_grid_load(image.width(), image.height()))
155    } else {
156        Ok(GridSize::new_xy(
157            image.width() / WIDTH as u32,
158            image.height() / HEIGHT as u32,
159        ))
160    }
161}

163/// Checks the size of the [`ImageBuffer`] before writing [`GridMap2D`] visual representation into it. Results in
164/// [`VisError`] if the image buffer size and [`GridSize`] does not match.
165pub fn check_grid_image_size<P: Pixel + 'static, const WIDTH: usize, const HEIGHT: usize>(
166    image: &ImageBuffer<P, Vec<P::Subpixel>>,
167    size: &GridSize,
168) -> Result<(), VisError<WIDTH, HEIGHT>> {
169    let expected = (size.x() * WIDTH as u32, size.y() * HEIGHT as u32);
170
171    if expected != (image.width(), image.height()) {
172        Err(VisError::new_grid_save(
173            expected,
174            (image.width(), image.height()),
175        ))
176    } else {
177        Ok(())
178    }
179}

183#[inline]
184pub(crate) fn create_tile_id_from_pixels<
185    P: PixelWithDefault,
186    const WIDTH: usize,
187    const HEIGHT: usize,
188>(
189    pixels: &[[P; WIDTH]; HEIGHT],
190) -> u64 {
191    let mut hasher = DefaultHasher::default();
192    pixels.hash(&mut hasher);
193    hasher.finish()
194}
