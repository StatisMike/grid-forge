1//! [`IdentifiableTileData`] instance builders.
2//!
3//! Many methods in `grid_forge` needs to construct new instances of [`GridMap2D`](crate::map::GridMap2D) and fill them with
4//! new instances of tiles by their `tile_id`. For them to be flexible, they need to use a builder struct, using some strategy
5//! to create new tile instances.
6//!
7//! User can create their own [`IdentTileBuilder`]-implementing struct to use their own method of building new tiles, though
8//! there are already some builders provided, using some basic strategies.
9
10use std::collections::BTreeMap;
11use std::error::Error;
12use std::fmt::Display;
13use std::marker::PhantomData;

15use crate::tile::identifiable::IdentifiableTileData;
16use crate::tile::{GridPosition, GridTile};

18/// [`IdentTileBuilder`] which creates new tiles of [`Clone`]-implementing tile struct. Prototype of tile with each `tile_id` need to be
19/// provided to the builder via [`add_tiles`](Self::add_tiles).
20///
21/// # Examples
22/// ```
23/// use grid_forge::{GridPosition, TileData, TileContainer};
24/// use grid_forge::identifiable::IdentifiableTileData;
25/// use grid_forge::identifiable::builders::{IdentTileBuilder, IdentTileCloneBuilder};
26///
27/// // Implementing custom TileData supported by `IdentTileCloneBuilder`.
28/// #[derive(Clone)]
29/// struct MyTileData {
30///     tile_type_id: u64,
31///     string: String
32/// }
33///
34/// impl TileData for MyTileData {};
35///
36/// impl IdentifiableTileData for MyTileData {
37///     fn tile_type_id(&self) -> u64 {
38///         self.tile_type_id
39///     }
40/// }
41///
42/// let mut builder = IdentTileCloneBuilder::<MyTileData>::default();
43/// let tiles = vec![
44///     MyTileData { tile_type_id: 1, string: "First".to_string() },
45///     MyTileData { tile_type_id: 2, string: "Second".to_string() },
46/// ];
47///
48/// builder.add_tiles(&tiles, false);
49///
50/// if let Err(err) = builder.check_missing_ids(&[1,2,3]) {
51///     assert_eq!(&[3], err.get_missing_tile_type_ids());
52/// } else {
53///     panic!("Should return error!");
54/// }
55///
56/// let tile_1st = builder.build_tile_unchecked(GridPosition::new_xy(2,3), 1);
57/// assert_eq!(
58///     (GridPosition::new_xy(2,3), 1, &"First".to_string()),
59///     (tile_1st.grid_position(), tile_1st.as_ref().tile_type_id, &tile_1st.as_ref().string)
60/// );
61///
62/// let tile_2nd = builder.build_tile_unchecked(GridPosition::new_xy(3,4), 2);
63/// assert_eq!(
64///     (GridPosition::new_xy(3,4), 2, &"Second".to_string()),
65///     (tile_2nd.grid_position(), tile_2nd.as_ref().tile_type_id, &tile_2nd.as_ref().string)
66/// );
67/// ```
68#[derive(Debug, Clone)]
69pub struct IdentTileCloneBuilder<Data: IdentifiableTileData + Clone> {
70    tiles: BTreeMap<u64, Data>,
71}
72
73impl<T: IdentifiableTileData + Clone> Default for IdentTileCloneBuilder<T> {
74    fn default() -> Self {
75        Self {
76            tiles: BTreeMap::new(),
77        }
78    }
79}
80
81impl<Data: IdentifiableTileData + Clone> IdentTileCloneBuilder<Data> {
82    /// Provide tile prototypes to the builder, which will be used to create new tile instances.
83    ///
84    /// If `overwrite` is `true`, then if prototype for given `tile_id` has been already saved, it will be overwritten.
85    pub fn add_tiles(&mut self, tiles: &[Data], overwrite: bool) {
86        for tile in tiles {
87            if !overwrite && self.tiles.contains_key(&tile.tile_type_id()) {
88                continue;
89            }
90            self.tiles.insert(tile.tile_type_id(), tile.clone());
91        }
92    }
93}

95impl<Data: IdentifiableTileData + Clone> IdentTileBuilder<Data> for IdentTileCloneBuilder<Data> {
96    fn build_tile_unchecked(&self, position: GridPosition, tile_type_id: u64) -> GridTile<Data> {
97        let tile_data = self
98            .tiles
99            .get(&tile_type_id)
100            .unwrap_or_else(|| panic!("can't get tile_data with `tile_type_id`: {tile_type_id}"))
101            .clone();
102
103        GridTile::new(position, tile_data)
104    }
105
106    fn build_tile(
107        &self,
108        position: GridPosition,
109        tile_type_id: u64,
110    ) -> Result<GridTile<Data>, TileBuilderError> {
111        if let Some(tile) = self.tiles.get(&tile_type_id) {
112            let data = tile.clone();
113            Ok(GridTile::new(position, data))
114        } else {
115            Err(TileBuilderError::new(&[tile_type_id]))
116        }
117    }
118
119    fn check_missing_ids(&self, tile_type_ids: &[u64]) -> Result<(), TileBuilderError> {
120        let missing_ids = tile_type_ids
121            .iter()
122            .filter(|tile_id| !self.tiles.contains_key(tile_id))
123            .copied()
124            .collect::<Vec<_>>();
125
126        if !missing_ids.is_empty() {
127            Err(TileBuilderError::new(&missing_ids))
128        } else {
129            Ok(())
130        }
131    }
132}

134/// [`IdentTileBuilder`] which creates new tiles with given identifier based on the contructor functions provided to the
135/// to the builder via [`set_tile_constructor`](Self::set_tile_constructor).
136///
137/// # Examples
138/// ```
139/// use grid_forge::{GridPosition, TileData, TileContainer};
140/// use grid_forge::identifiable::IdentifiableTileData;
141/// use grid_forge::identifiable::builders::{IdentTileBuilder, IdentTileFunBuilder};
142///
143/// // Implementing custom TileData supported by `IdentTileCloneBuilder`.
144/// #[derive(Clone)]
145/// struct MyTileData {
146///     tile_type_id: u64,
147///     traversible: bool
148/// }
149///
150/// impl TileData for MyTileData {};
151///
152/// impl IdentifiableTileData for MyTileData {
153///     fn tile_type_id(&self) -> u64 {
154///         self.tile_type_id
155///     }
156/// }
157///
158/// let mut builder = IdentTileFunBuilder::<MyTileData>::default();
159/// builder.set_tile_constructor(1, ||( MyTileData { tile_type_id: 1, traversible: true} ));
160/// builder.set_tile_constructor(2, ||( MyTileData { tile_type_id: 2, traversible: false} ));
161///
162/// if let Err(err) = builder.check_missing_ids(&[1,2,3]) {
163///     assert_eq!(&[3], err.get_missing_tile_type_ids());
164/// } else {
165///     panic!("Should return error!");
166/// }
167///
168/// let tile_1st = builder.build_tile_unchecked(GridPosition::new_xy(2,3), 1);
169/// assert_eq!((GridPosition::new_xy(2,3), 1, true), (tile_1st.grid_position(), tile_1st.as_ref().tile_type_id(), tile_1st.as_ref().traversible));
170///
171/// let tile_2nd = builder.build_tile_unchecked(GridPosition::new_xy(3,4), 2);
172/// assert_eq!((GridPosition::new_xy(3,4), 2, false), (tile_2nd.grid_position(), tile_2nd.as_ref().tile_type_id(), tile_2nd.as_ref().traversible));
173/// ```
174#[derive(Debug, Clone)]
175pub struct IdentTileFunBuilder<T: IdentifiableTileData> {
176    funs: BTreeMap<u64, fn() -> T>,
177}
178
179impl<Data: IdentifiableTileData> IdentTileFunBuilder<Data> {
180    pub fn set_tile_constructor(&mut self, tile_id: u64, constructor: fn() -> Data) {
181        self.funs.insert(tile_id, constructor);
182    }
183
184    pub fn clear(&mut self) {
185        self.funs.clear();
186    }
187}
188
189impl<Data: IdentifiableTileData> Default for IdentTileFunBuilder<Data> {
190    fn default() -> Self {
191        Self {
192            funs: BTreeMap::new(),
193        }
194    }
195}

197impl<Data: IdentifiableTileData> IdentTileBuilder<Data> for IdentTileFunBuilder<Data> {
198    fn build_tile_unchecked(&self, position: GridPosition, tile_type_id: u64) -> GridTile<Data> {
199        let fun = self.funs.get(&tile_type_id).unwrap_or_else(|| {
200            panic!("can't get tile constructor function for `tile_type_id`: {tile_type_id}")
201        });
202
203        GridTile::new(position, fun())
204    }
205
206    fn build_tile(
207        &self,
208        position: GridPosition,
209        tile_id: u64,
210    ) -> Result<GridTile<Data>, TileBuilderError> {
211        if let Some(fun) = self.funs.get(&tile_id) {
212            Ok(GridTile::new(position, fun()))
213        } else {
214            Err(TileBuilderError::new(&[tile_id]))
215        }
216    }
217
218    fn check_missing_ids(&self, tile_ids: &[u64]) -> Result<(), TileBuilderError> {
219        let missing_ids = tile_ids
220            .iter()
221            .filter(|tile_id| !self.funs.contains_key(tile_id))
222            .copied()
223            .collect::<Vec<_>>();
224
225        if !missing_ids.is_empty() {
226            Err(TileBuilderError::new(&missing_ids))
227        } else {
228            Ok(())
229        }
230    }
231}

233/// Trait which allows creating new istance of struct implementing [`IdentifiableTileData`].
234///
235/// See also [`BasicIdentTileData`](crate::tile::identifiable::BasicIdentTileData) for basic identifiable tile type which
236/// implements this trait out of the box whenever you don't need your tile struct to hold any additional information.
237///
238/// # Examples
239///
240/// Implementing this trait for you custom tile makes it constructable via [`IdentTileTraitBuilder`].
241///
242/// ```
243/// use grid_forge::{GridPosition, TileData, TileContainer};
244/// use grid_forge::identifiable::IdentifiableTileData;
245/// use grid_forge::identifiable::builders::{ConstructableViaIdentifierTile, IdentTileBuilder, IdentTileTraitBuilder};
246///
247/// // Implementing custom TileData supported by `IdentTileCloneBuilder`.
248/// #[derive(Clone)]
249/// struct MyTileData {
250///     tile_type_id: u64,
251/// }
252///
253/// impl TileData for MyTileData {};
254///
255/// impl IdentifiableTileData for MyTileData {
256///     fn tile_type_id(&self) -> u64 {
257///         self.tile_type_id
258///     }
259/// }
260///
261/// impl ConstructableViaIdentifierTile for MyTileData {
262///     fn tile_new(tile_type_id: u64) -> Self {
263///         Self { tile_type_id }
264///     }
265/// }
266///
267/// let builder = IdentTileTraitBuilder::<MyTileData>::default();
268/// let tile = builder.build_tile_unchecked(GridPosition::new_xy(2,3), 45);
269///
270/// assert_eq!(GridPosition::new_xy(2,3), tile.grid_position());
271/// assert_eq!(45, tile.as_ref().tile_type_id());
272/// ```
273pub trait ConstructableViaIdentifierTile
274where
275    Self: IdentifiableTileData,
276{
277    fn tile_new(tile_type_id: u64) -> Self;
278}

280/// [`IdentTileBuilder`] which creates new tiles with given identifier based on the tile implementation of
281/// [`ConstructableViaIdentifierTile`].
282///
283/// There's no need to add any tile creators, as it is purely based on implemented traits, so [`IdentTileBuilder::check_missing_ids`]
284/// is an no-op for this struct.
285///
286/// Refer to documentation of [`ConstructableViaIdentifierTile`] for usage example.
287#[derive(Debug, Clone)]
288pub struct IdentTileTraitBuilder<Data: IdentifiableTileData + ConstructableViaIdentifierTile> {
289    phantom: PhantomData<Data>,
290}
291
292impl<Data: IdentifiableTileData + ConstructableViaIdentifierTile> Default
293    for IdentTileTraitBuilder<Data>
294{
295    fn default() -> Self {
296        Self {
297            phantom: PhantomData::<Data>,
298        }
299    }
300}
301
302impl<Data: IdentifiableTileData + ConstructableViaIdentifierTile> IdentTileBuilder<Data>
303    for IdentTileTraitBuilder<Data>
304{
305    fn build_tile_unchecked(&self, position: GridPosition, tile_type_id: u64) -> GridTile<Data> {
306        GridTile::new(position, Data::tile_new(tile_type_id))
307    }
308
309    fn build_tile(
310        &self,
311        position: GridPosition,
312        tile_type_id: u64,
313    ) -> Result<GridTile<Data>, TileBuilderError> {
314        Ok(self.build_tile_unchecked(position, tile_type_id))
315    }
316
317    fn check_missing_ids(&self, _tile_ids: &[u64]) -> Result<(), TileBuilderError> {
318        Ok(())
319    }
320}

322/// Trait shared by objects which on basis of the grid position and tile identifier of given [`IdentifiableTileData`]-implementing struct can
323/// create correct instance of the tile. Necessary for many [`GridMap2D`](crate::map::GridMap2D) creating methods.
324///
325/// Three different builders are available in the `grid_forge`:
326/// - [`IdentTileFunBuilder`] - for tiles not implementing any additional traits.
327/// - [`IdentTileCloneBuilder`] - for tiles implementing [`Clone`].
328/// - [`IdentTileTraitBuilder`] - for tiles implementing [`ConstructableViaIdentifierTile`].
329///
330/// The logic for building tile is encapsulated in [`build_tile_unchecked`](IdentTileBuilder::build_tile_unchecked) and
331/// [`build_tile`](IdentTileBuilder::build_tile) methods. The `unchecked` version is usually faster and is recommended
332/// to be used when creating a batch of tiles at once - can be used reliably if [`check_missing_ids`](IdentTileBuilder::check_missing_ids)
333/// is called before beginning the batch operation.
334pub trait IdentTileBuilder<Data: IdentifiableTileData> {
335    /// Creates tile with given tile identifier at given grid position.
336    ///
337    /// # Panics
338    /// Can panic if builder does not have possibility to construct tile of given `tile_id` based on the gathered information. You can check
339    /// for missing tile ids with [`check_missing_ids`](IdentTileBuilder::check_missing_ids) or use its fallible version:
340    /// [`build_tile`](IdentTileBuilder::build_tile).
341    fn build_tile_unchecked(&self, position: GridPosition, tile_type_id: u64) -> GridTile<Data>;
342
343    /// Creates tile with given tile identifier at given grid position. Returns error if cannot construct tile of given `tile_id`.
344    fn build_tile(
345        &self,
346        position: GridPosition,
347        tile_type_id: u64,
348    ) -> Result<GridTile<Data>, TileBuilderError>;
349
350    /// Checks for missing tile creators out of provided slice of `tile_id`.
351    fn check_missing_ids(&self, tile_type_ids: &[u64]) -> Result<(), TileBuilderError>;
352}
353
354/// Error stemming from missing tiles in [`IdentTileBuilder`].
355#[derive(Debug, Clone)]
356pub struct TileBuilderError {
357    tile_type_ids: Vec<u64>,
358}
359
360impl TileBuilderError {
361    fn new(tile_ids: &[u64]) -> Self {
362        Self {
363            tile_type_ids: Vec::from(tile_ids),
364        }
365    }
366
367    pub fn get_missing_tile_type_ids(&self) -> &[u64] {
368        &self.tile_type_ids
369    }
370}
371
372impl Display for TileBuilderError {
373    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
374        write!(
375            f,
376            "missing tile ids from builder: {missing_ids}",
377            missing_ids = self
378                .tile_type_ids
379                .iter()
380                .map(|f| f.to_string())
381                .collect::<Vec<_>>()
382                .join(", ")
383        )
384    }
385}
